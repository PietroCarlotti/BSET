\section{Derivation of \texorpdfstring{\(\theta\)}{theta} and \texorpdfstring{\(\delta\)}{delta} as Expectations}
\label{app:derivation_theta_delta}

This appendix derives expressions for \(\theta\) and \(\delta\) as expectations over the covariate distribution. 

\subsection{Derivation of \texorpdfstring{\(\theta\)}{theta} as an Expectation over the Covariate Distribution}
\label{app:derivation_theta}

Under the model in Equation~\eqref{eq:data_generating_process_covariates}, it is easy to see that
\begin{equation*}
    Y_{1i} - Y_{0i} \mid X_{i} = x \overset{\text{\footnotesize ind.}}{\sim} \mathcal{N} \left( \frac{ \mu^{x}_{Y_{1}} - \mu^{x}_{Y_{0}}}{\sqrt{\Sigma^{x}_{Y_{1} Y_{1}} + \Sigma^{x}_{Y_{0} Y_{0}} - 2 \Sigma^{x}_{Y_{1} Y_{0}}}} \right), \quad i = 1, \ldots, n, \; \forall \; x \in \mathcal{X}.
\end{equation*}
Therefore, we obtain this analytical expression for \(V^{x}_{Y}\):
\begin{align*}
        V^{x}_{Y} & = \Prob \left( Y_{1i} > Y_{0i} \mid X_{i} = x \right) \\
        & = \Prob \left( Y_{1i} - Y_{0i} > 0 \mid X_{i} = x \right) \\
        & = \Prob \left( \frac{Y_{1i} - Y_{0i} - \mu^{x}_{Y_{1}} + \mu^{x}_{Y_{0}}}{\sqrt{\Sigma^{x}_{Y_{1} Y_{1}} + \Sigma^{x}_{Y_{0} Y_{0}} - 2 \Sigma^{x}_{Y_{1} Y_{0}}}} > \frac{- \mu^{x}_{Y_{1}} + \mu^{x}_{Y_{0}}}{\sqrt{\Sigma^{x}_{Y_{1} Y_{1}} + \Sigma^{x}_{Y_{0} Y_{0}} - 2 \Sigma^{x}_{Y_{1} Y_{0}}}} \mid X_{i} = x \right) \\
        & = \Prob \left( W > \frac{- \mu^{x}_{Y_{1}} + \mu^{x}_{Y_{0}}}{\sqrt{\Sigma^{x}_{Y_{1} Y_{1}} + \Sigma^{x}_{Y_{0} Y_{0}} - 2 \Sigma^{x}_{Y_{1} Y_{0}}}} \right) \\
        & = 1 - \Phi \left( \frac{- \mu^{x}_{Y_{1}} + \mu^{x}_{Y_{0}}}{\sqrt{\Sigma^{x}_{Y_{1} Y_{1}} + \Sigma^{x}_{Y_{0} Y_{0}} - 2 \Sigma^{x}_{Y_{1} Y_{0}}}} \right) \\
        & = \Phi \left( \frac{ \mu^{x}_{Y_{1}} - \mu^{x}_{Y_{0}}}{\sqrt{\Sigma^{x}_{Y_{1} Y_{1}} + \Sigma^{x}_{Y_{0} Y_{0}} - 2 \Sigma^{x}_{Y_{1} Y_{0}}}} \right), \; \forall \; x \in \mathcal{X},
\end{align*}
where \(W \sim \mathcal{N}(0,1)\) and \(\Phi\) is the standard normal CDF. Then, by the law of total probability, we can derive an analytical expression for \(V_{Y}\) as follows:
\begin{align*}
    V_{Y} & = \Prob \left( Y_{1i} > Y_{0i} \right) \\
    & = \int_{\mathcal{X}} V^{x}_{Y} \, f (x) \, \text{d}x \\
    & = \int_{\mathcal{X}} \Phi \left( \frac{ \mu^{x}_{Y_{1}} - \mu^{x}_{Y_{0}}}{\sqrt{\Sigma^{x}_{Y_{1} Y_{1}} + \Sigma^{x}_{Y_{0} Y_{0}} - 2 \Sigma^{x}_{Y_{1} Y_{0}}}} \right) \, f (x) \, \text{d}x.
\end{align*}
Applying a similar logic to \(V_S\), we obtain this analytical expression for \(\theta\):
\begin{equation*}
    \theta = \int_{\mathcal{X}} \theta^{x} \, f(x) \, \text{d}x,
\end{equation*}
where we denote
\begin{equation*}
    \theta^{x} = %V^{x}_{Y} - V^{x}_{S} =
    \Phi \left( \frac{ \mu^{x}_{Y_{1}} - \mu^{x}_{Y_{0}}}{\sqrt{\Sigma^{x}_{Y_{1} Y_{1}} + \Sigma^{x}_{Y_{0} Y_{0}} - 2 \Sigma^{x}_{Y_{1} Y_{0}}}} \right) - \Phi \left( \frac{ \mu^{x}_{S_{1}} - \mu^{x}_{S_{0}}}{\sqrt{\Sigma^{x}_{S_{1} S_{1}} + \Sigma^{x}_{S_{0} S_{0}} - 2 \Sigma^{x}_{S_{1} S_{0}}}} \right), \; \forall \; x \in \mathcal{X}.
\end{equation*}

%\begin{equation}
%\resizebox{.91\textwidth}{!}{%
%    \( \displaystyle \theta = \int_{\mathcal{X}} \left[ \Phi \left( \frac{ \mu^{x}_{Y_{1}} - \mu^{x}_{Y_{0}}}{\sqrt{\Sigma^{x}_{Y_{1} Y_{1}} + \Sigma^{x}_{Y_{0} Y_{0}} - 2 \Sigma^{x}_{Y_{1} Y_{0}}}} \right) - \Phi \left( \frac{ \mu^{x}_{S_{1}} - \mu^{x}_{S_{0}}}{\sqrt{\Sigma^{x}_{S_{1} S_{1}} + \Sigma^{x}_{S_{0} S_{0}} - 2 \Sigma^{x}_{S_{1} S_{0}}}} \right) \right] f (x) \, \text{d}x \)

\subsection{Derivation of \texorpdfstring{\(\delta\)}{delta} as an Expectation over the Covariate Distribution}
\label{app:derivation_delta}

Under the model in Equation~\eqref{eq:data_generating_process_covariates}, it is easy to see that
\begin{equation*}
    Y_{1i} - Y_{0j} \mid X_{i} = x_{i}, X_{j} = x_{j} \sim \mathcal{N} \left( \frac{\mu^{x_{i}}_{Y_{1}} - \mu^{x_{j}}_{Y_{0}}}{\sqrt{\Sigma^{x_{i}}_{Y_{1} Y_{1}} + \Sigma^{x_{j}}_{Y_{0} Y_{0}}}} \right), \; \forall \; x_{i}, x_{j} \in \mathcal{X}, \; i,j = 1, \ldots, n.
\end{equation*}
Therefore, we obtain this analytical expression for \(U^{x_{i} x_{j}}_{Y}\):
\begin{align*}
        U^{x_{i} x_{j}}_{Y} & = \Prob \left( Y_{1i} > Y_{0j} \mid X_{i} = x_{i}, X_{j} = x_{j} \right) \\
        & = \Prob \left( \frac{Y_{1i} - Y_{0j} - \mu^{x_{i}}_{Y_{1}} + \mu^{x_{j}}_{Y_{0}}}{\sqrt{\Sigma^{x_{i}}_{Y_{1} Y_{1}} + \Sigma^{x_{j}}_{Y_{0} Y_{0}}}} > \frac{- \mu^{x_{i}}_{Y_{1}} + \mu^{x_{j}}_{Y_{0}}}{\sqrt{\Sigma^{x_{i}}_{Y_{1} Y_{1}} + \Sigma^{x_{j}}_{Y_{0} Y_{0}}}} \mid X_{i} = x_{i}, X_{j} = x_{j} \right) \\
        & = \Prob \left( W > \frac{- \mu^{x_{i}}_{Y_{1}} + \mu^{x_{j}}_{Y_{0}}}{\sqrt{\Sigma^{x_{i}}_{Y_{1} Y_{1}} + \Sigma^{x_{j}}_{Y_{0} Y_{0}}}} \right) \\
        & = 1 - \Phi \left( \frac{- \mu^{x_{i}}_{Y_{1}} + \mu^{x_{j}}_{Y_{0}}}{\sqrt{\Sigma^{x_{i}}_{Y_{1} Y_{1}} + \Sigma^{x_{j}}_{Y_{0} Y_{0}}}} \right) \\
        & = \Phi \left( \frac{\mu^{x_{i}}_{Y_{1}} - \mu^{x_{j}}_{Y_{0}}}{\sqrt{\Sigma^{x_{i}}_{Y_{1} Y_{1}} + \Sigma^{x_{j}}_{Y_{0} Y_{0}}}} \right), \; \forall \; x_{i}, x_{j} \in \mathcal{X}, \; i,j = 1, \ldots, n.
\end{align*}
Then, by the law of total probability, we can derive an analytical expression for \(U_{Y}\) as follows:
\begin{align*}
        U_{Y} & = \Prob \left( Y_{1i} > Y_{0j} \right) \\
        & = \int_{\mathcal{X}} \int_{\mathcal{X}} U^{x_{i}, x_{j}}_{Y} \, f (x_{i}) \, f (x_{j}) \, \text{d}x_{i} \, \text{d}x_{j} \\
        & = \int_{\mathcal{X}} \int_{\mathcal{X}} \Phi \left( \frac{\mu^{x_{i}}_{Y_{1}} - \mu^{x_{j}}_{Y_{0}}}{\sqrt{\Sigma^{x_{i}}_{Y_{1} Y_{1}} + \Sigma^{x_{j}}_{Y_{0} Y_{0}}}} \right) \, f (x_{i}) \, f (x_{j}) \, \text{d}x_{i} \, \text{d}x_{j}.
\end{align*}
Applying a similar logic to \(U_{S}\), we obtain this analytical expression for \(\delta\):
\begin{equation*}
    \delta = \int_{\mathcal{X}} \int_{\mathcal{X}} \delta^{x_{i} x_{j}} \, f(x_{i}) \, f(x_{j}) \, \text{d}x_{i} \, \text{d}x_{j},
\end{equation*}
where we denote
\begin{equation*}
    \delta^{x_{i} x_{j}} = %U^{x_{i} x_{j}}_{Y} - U^{x_{i} x_{j}}_{S} =
    \Phi \left( \frac{\mu^{x_{i}}_{Y_{1}} - \mu^{x_{j}}_{Y_{0}}}{\sqrt{\Sigma^{x_{i}}_{Y_{1} Y_{1}} + \Sigma^{x_{j}}_{Y_{0} Y_{0}}}} \right) - \Phi \left( \frac{\mu^{x_{i}}_{S_{1}} - \mu^{x_{j}}_{S_{0}}}{\sqrt{\Sigma^{x_{i}}_{S_{1} S_{1}} + \Sigma^{x_{j}}_{S_{0} S_{0}}}} \right), \; \forall \; x_i, x_j \in \mathcal{X}.
\end{equation*}

\subsection{Derivation in the Condionally Orthogonal and Homoscedastic Setting}

Under the assumption of conditional orthogonality stated in the paper, it follows that
\begin{equation*}
    V^{x}_{Y} = U^{xx}_{Y} \text{ and } V^{x}_{S} = U^{xx}_{S}, \; \forall \; x \in \mathcal{X},
\end{equation*}
so that
\begin{equation}
\label{eq:equivalence_theta_x_delta_xx}
    \theta^{x} = \delta^{x x}, \; \forall \; x \in \mathcal{X}.
\end{equation}
Under the assumption of conditional homoscedasticity stated in the paper, it follows that
\begin{equation*}
    \theta = \E_{X_{i} \sim f} \!\left[ g(X_{i}, X_{i}) \right], \quad \delta = \E_{(X_{i}, X_{j}) \sim f^{2}} \!\left[ g(X_{i}, X_{j}) \right],
\end{equation*}
where we denote
\begin{equation}
\label{eq:g_definition}
    g(x_{i}, x_{j}) = \Phi \left( \mu^{x_{i}}_{Y_{1}} - \mu^{x_{j}}_{Y_{0}} \right) - \Phi \left( \mu^{x_{i}}_{S_{1}} - \mu^{x_{j}}_{S_{0}} \right), \; \forall \; x_{i}, x_{j} \in \mathcal{X}.
\end{equation}

\section{Derivation of \texorpdfstring{\(|\theta - \delta|\)}{|theta - delta|} in the Binary Covariate Setting}
\label{app:binary_covariate_derivation}

Under the setting described in Section~\ref{subsec:binary_covariate_setting} of the main paper, \(\theta\) and \(\delta\) reduce to
\begin{equation*}
    \theta = \frac{\theta^{0} + \theta^{1}}{2}, \quad \delta = \frac{\delta^{00} + \delta^{11} + \delta^{01} + \delta^{10}}{4}.
\end{equation*}
Based on Equation~\eqref{eq:equivalence_theta_x_delta_xx}, we have that
\begin{equation}
\label{eq:distance_theta_delta_binary_covariate}
    | \theta - \delta | = \frac{ \left| \delta^{00} + \delta^{11} - \delta^{01} - \delta^{10} \right| }{4}.
\end{equation}
This expression highlights that the distance between the two parameters is driven by differences between comparisons within the same covariate group and comparisons across different covariate groups. Importantly, this effect does not arise from treatment effect heterogeneity.

To derive a formula for \(|\theta - \delta|\), let us further simplify the setting by assuming that \(S\) is a perfect surrogate for \(Y\) with homogeneous treatment effects across covariate groups:
\begin{equation*}
    \mu^{x}_{Y_{1}} - \mu^{x}_{Y_{0}} =
    \mu^{x}_{S_{1}} - \mu^{x}_{S_{0}} = \Delta > 0, \; \forall \; x \in \{0,1\}.
\end{equation*}
In this setting, we can easily compute the terms in Equation~\eqref{eq:distance_theta_delta_binary_covariate} using Equation~\eqref{eq:g_definition}:
\begin{align*}
        \delta^{00} & = \Phi(\Delta) - \Phi(\Delta) = 0, \\
        \delta^{11} & = \Phi(\Delta) - \Phi(\Delta) = 0, \\
        \delta^{01} & = \Phi\!\left(\Delta + \mu^{0}_{Y_{0}} - \mu^{1}_{Y_{0}}\right) - \Phi\!\left(\Delta + \mu^{0}_{S_{0}} - \mu^{1}_{S_{0}}\right), \\
        \delta^{10} & = \Phi\!\left(\Delta + \mu^{1}_{Y_{0}} - \mu^{0}_{Y_{0}}\right) - \Phi\!\left(\Delta + \mu^{1}_{S_{0}} - \mu^{0}_{S_{0}}\right).
\end{align*}
Hence, we conclude that 
\begin{equation*}
    |\theta - \delta| = \frac{1}{4} \left| \Phi(\Delta + d_{Y}) - \Phi(\Delta + d_{S}) + \Phi(\Delta - d_{Y}) - \Phi(\Delta - d_{S}) \right|,
\end{equation*}
where we denote \(d_{Y} = \mu^{0}_{Y_{0}} - \mu^{1}_{Y_{0}}\) and \(d_{S} = \mu^{0}_{S_{0}} - \mu^{1}_{S_{0}}\).

\subsection{Supremum of \texorpdfstring{\(|\theta - \delta|\)}{|theta - delta|}}
\label{app:supremum_delta_binary_covariate}

We now compute analytically the supremum of \(|\theta - \delta|\) with respect to \(d_{Y}\) and \(d_{S}\). First, observe that the distance \(|\theta - \delta|\) can be rewritten as
\begin{equation*}
    |\theta - \delta| = \frac{1}{4} \left| g_{\Delta}(d_{Y}) - g_{\Delta}(d_{S}) \right|,
\end{equation*}
where 
\begin{equation*}
    g_{\Delta}(x) = \Phi(\Delta + x) + \Phi(\Delta - x), \; \forall \; x \in \mathbb{R}.
\end{equation*}
Then, it follows that
\begin{equation*}
    \sup_{d_{Y}, d_{S}} \{ |\theta - \delta| \} = \frac{1}{4} \left( \sup_{x} \{ g_{\Delta}(x) \} - \inf_{x} \{ g_{\Delta}(x) \} \right)
\end{equation*}
Moreover, since 
\begin{equation*}
    g_{\Delta}(-x) = g_{\Delta}(x), \; \forall \; x \in \mathbb{R},
\end{equation*}
we only have to optimize \(g_{\Delta}\) over \(\R_{+}\). Observe that the derivative of \(g_{\Delta}\) is given by
\begin{align*}
    g'_{\Delta}(x) & = \phi(\Delta + x) - \phi(\Delta - x) \\
    & = \frac{1}{\sqrt{2 \pi}} \left( e^{-\frac{(\Delta + x)^2}{2}} - e^{-\frac{(\Delta - x)^2}{2}} \right), \; \forall \; x \in \R_{+},
\end{align*}
where \(\phi\) is the standard normal PDF. Then, the critical points of \(g_{\Delta}\) satisfy this equation:
\begin{equation*}
    \phi(\Delta + x) = \phi(\Delta - x) \iff ( \Delta + x )^2 = ( \Delta - x )^2,
\end{equation*}
which has only one solution \(x = 0\), since we assumed that \(\Delta > 0\). Moreover, we have that
\begin{equation*}
    g'_{\Delta}(x) < 0, \; \forall \; x \in \R_{+}.
\end{equation*}
This implies that \(g_{\Delta}\) is strictly decreasing over \(\R_{+}\), so that
\begin{align*}
    \sup_{x} \{ g_{\Delta}(x) \} & = g_{\Delta}(0) = 2 \Phi(\Delta), \\
    \inf_{x} \{ g_{\Delta}(x) \} & = \lim_{x \to +\infty} g_{\Delta}(x) = 1.
\end{align*}
Hence, we conclude that
\begin{equation*}
    \sup_{d_{Y}, d_{S}} \{ |\theta - \delta| \} = \frac{1}{4} \left( 2 \Phi(\Delta) - 1 \right) = \frac{1}{2} \Phi(\Delta) - \frac{1}{4}.
\end{equation*}

\section{Derivation of \texorpdfstring{\(\theta\)}{theta} and \texorpdfstring{\(\delta\)}{delta} in the Gaussian Covariate Setting}
\label{app:gaussian_covariate_derivation}

Using the setting from Section~\ref{subsec:gaussian_covariate_setting}, we express \(\theta\) and \(\delta\) via Equation~\eqref{eq:g_definition} as follows:
\begin{align*}
    \theta & = \E_{X_{i} \sim f} \!\left[ \Phi \left( X_{i} (\beta_{Y_{1}} - \beta_{Y_{0}}) \right) - \Phi \left( X_{i} (\beta_{S_{1}} - \beta_{S_{0}}) \right) \right], \\
    \delta & = \E_{(X_{i}, X_{j}) \sim f^{2}} \!\left[ \Phi \left( X_{i} \beta_{Y_{1}} - X_{j} \beta_{Y_{0}} \right) - \Phi \left( X_{i} \beta_{S_{1}} - X_{j} \beta_{S_{0}} \right) \right].
\end{align*}

\subsection{Derivation of \texorpdfstring{\(\theta\)}{theta}}
\label{app:derivation_theta_gaussian_covariate}

Using the perfect surrogacy assumption in Equation~\eqref{eq:perfect_surrogacy_assumption_gaussian_covariate}, we have that
\begin{align*}
    \theta & = \E_{X_{i} \sim f} \!\left[ \Phi \left( X_{i} (\beta_{Y_{1}} - \beta_{Y_{0}}) \right) - \Phi \left( X_{i} (\beta_{S_{1}} - \beta_{S_{0}}) \right) \right] \\
    & = \E_{X_{i} \sim f} \!\left[ \Phi \left( X_{i} \Delta \right) - \Phi \left( X_{i} \Delta \right) \right] = 0.
\end{align*}

\subsection{Derivation of \texorpdfstring{\(\delta\)}{delta}}
\label{app:derivation_delta_gaussian_covariate}

Observe that by the Gaussian covariate assumption, we have that
\begin{align*}
    X_{i} \beta_{Y_{1}} - X_{j} \beta_{Y_{0}} & \sim \mathcal{N} \left( m (\beta_{Y_{1}} - \beta_{Y_{0}}), v^2 (\beta_{Y_{1}}^2 + \beta_{Y_{0}}^2) \right), \quad i,j = 1, \ldots, n, \\
    X_{i} \beta_{S_{1}} - X_{j} \beta_{S_{0}} & \sim \mathcal{N} \left( m (\beta_{S_{1}} - \beta_{S_{0}}), v^2 (\beta_{S_{1}}^2 + \beta_{S_{0}}^2) \right), \quad i,j = 1, \ldots, n.
\end{align*}
Therefore, we can derive an expanded analytical expression for \(\delta\) as follows:
\begin{align*}
    \delta & = \E_{(X_{i}, X_{j}) \sim f^{2}} \!\left[ \Phi \left( X_{i} \beta_{Y_{1}} - X_{j} \beta_{Y_{0}} \right) \right] - \E_{(X_{i}, X_{j}) \sim f^{2}} \!\left[ \Phi \left( X_{i} \beta_{S_{1}} - X_{j} \beta_{S_{0}} \right) \right] \\
    & = \Phi \left( \frac{m [\beta_{Y_{1}} - \beta_{Y_{0}}]}{\sqrt{1 + v^2 [\beta_{Y_{1}}^2 + \beta_{Y_{0}}^2]}} \right) - \Phi \left( \frac{m [\beta_{S_{1}} - \beta_{S_{0}}]}{\sqrt{1 + v^2 [\beta_{S_{1}}^2 + \beta_{S_{0}}^2]}} \right),
\end{align*}
where the last line follows from the following property of Gaussian random variables.

\begin{lemma}
\label{lemma:gaussian_phi_identity}
    Let \(W \sim \mathcal{N}(m, v^2)\) for some \(m \in \mathbb{R}\) and \(v \in \R_{+}\). Then, we have that
    \begin{equation*}
        \E_{W \sim \mathcal{N}(m, v^2)} \!\left[ \Phi(W) \right] = \Phi \!\left( \frac{m}{\sqrt{1 + v^2}} \right).
    \end{equation*}
\end{lemma}

\subsection{Supremum of \texorpdfstring{\(|\theta - \delta|\)}{|theta - delta|}}
\label{app:supremum_delta_gaussian_covariate}

We now compute analytically the supremum of \(|\theta - \delta|\) with respect to \(\boldsymbol{\beta}\). First, observe that under the perfect surrogacy assumption in Equation~\eqref{eq:perfect_surrogacy_assumption_gaussian_covariate}, we have that
\begin{align*}
    |\theta - \delta| & = |\delta| \\
    & = \left| \Phi \left( \frac{m \Delta}{\sqrt{1 + v^2 [(\beta_{Y_{0}} + \Delta)^2 + \beta_{Y_{0}}^2]}} \right) - \Phi \left( \frac{m \Delta}{\sqrt{1 + v^2 [(\beta_{S_{0}} + \Delta)^2 + \beta_{S_{0}}^2]}} \right) \right| \\
    & = | h_{\Delta, m, v} (\beta_{Y_{0}}) - h_{\Delta, m, v}(\beta_{S_{0}}) |,
\end{align*}
where
\begin{equation*}
    h_{\Delta, m, v}(x) = \Phi \left( \frac{m \Delta}{\sqrt{1 + v^2 [(x + \Delta)^2 + x^2]}} \right), \; \forall \; x \in \mathbb{R}.
\end{equation*}
Then, it follows that
\begin{equation*}
    \sup_{\boldsymbol{\beta}} \{ |\theta - \delta| \} = \sup_{x} \{ h_{\Delta, m, v}(x) \} - \inf_{x} \{ h_{\Delta, m, v}(x) \}.
\end{equation*}
Notice that since \(x\) only appears in the denominator of the argument of \(\Phi\), finding the extrema of \(h_{\Delta, m, v}\) is equivalent to finding the extrema of the function \(k_{\Delta, m, v}\) defined as follows:
\begin{equation*}
    k_{\Delta, m, v}(x) = 1 + v^2 [(x + \Delta)^2 + x^2] = 1 + v^2 (2 x^2 + 2 \Delta x + \Delta^2), \; \forall \; x \in \mathbb{R}.
\end{equation*}
It is easy to see that
\begin{align*}
    \sup_{x} \{ k_{\Delta, m, v}(x) \} & = \lim_{x \to +\infty} k_{\Delta, m, v}(x) = +\infty, \\
    \inf_{x} \{ k_{\Delta, m, v}(x) \} & = k_{\Delta, m, v}\!\left(-\frac{\Delta}{2}\right) = 1 + \frac{v^2 \Delta^2}{2},
\end{align*}
where we omit the algebraic details for brevity. Therefore, we have that
\begin{align*}
    \sup_{\boldsymbol{\beta}} \{ |\theta - \delta| \} & = \left| \Phi \left( \frac{m \Delta}{\sqrt{1 + \frac{v^2 \Delta^2}{2}}} \right) - \Phi \left( 0 \right) \right| \\
    & = \left| \Phi \left( \frac{m}{\sqrt{\frac{1}{\Delta^2} + \frac{v^2}{2}}} \right) - \frac{1}{2} \right|,
\end{align*}
which is strictly increasing in \( | \Delta | \), so that
\begin{align*}
    \sup_{\boldsymbol{\beta}, \Delta} \{ |\theta - \delta| \} & = \lim_{|\Delta| \to +\infty} \left| \Phi \left( \frac{m}{\sqrt{\frac{1}{\Delta^2} + \frac{v^2}{2}}} \right) - \frac{1}{2} \right| \\
    & = \left| \Phi \left( \frac{m \, \sqrt{2}}{v} \right) - \frac{1}{2} \right|.
\end{align*}

\section{Regularity and Posterior Consistency}
\label{app:posterior_consistency}

This appendix provides a formal definition of regularity for Bayesian models and states Schwartz's theorem, which is a fundamental result on posterior consistency of regular models.

\begin{definition}[Regularity of a Bayesian model]
\label{def:regularity}
  Let a Bayesian model be defined as
  \begin{align*}
      D_{i} \mid \phi & \iid f_{\phi}, \quad i = 1, \ldots, n, \\
      \phi & \sim \pi,
  \end{align*}
  where \(D_{i}\) is the observed data for unit \(i\), \(f_{\phi}\) is the likelihood parametrized by \(\phi\), and \(\pi\) is a prior distribution over \(\phi\). Then, the model is \textit{regular} if the following two conditions hold:
  \begin{itemize}[leftmargin=*]
    \item \textbf{KL support of the prior}: \(\pi\) assigns positive mass to every KL neighborhood of \(\phi^{*}\):
    \begin{equation*}
      \pi \left( D_{\mathrm{KL}}(f_{\phi^{*}} \| f_{\phi}) < \varepsilon \right) > 0, \; \forall \; \varepsilon \in \R_{+},
    \end{equation*}
    where \(D_{\mathrm{KL}}(f_{\phi^{*}} \| f_{\phi})\) is the KL divergence of \(f_{\phi}\) from \(f_{\phi^{*}}\).
    \item \textbf{Identifiability}: The model is identifiable: 
    \begin{equation*}
        f_{\phi} \neq f_{\phi'}, \; \forall \; \phi \neq \phi'.
    \end{equation*}
  \end{itemize}
\end{definition}

\begin{theorem}[Schwartz's theorem \citep{schwartz1965bayes}]
  Let the true DGP be defined as
  \begin{equation*}
    D_{i} \iid f_{\phi^{*}}, \quad i = 1, \ldots, n,
  \end{equation*}
  where \(D_{i}\) is the observed data for unit \(i\) and \(f_{\phi^{*}}\) is the true distribution parametrized by \(\phi^{*}\). Let the Bayesian model be defined as
  \begin{align*}
      D_{i} \mid \phi & \iid f_{\phi}, \quad i = 1, \ldots, n, \\
      \phi & \sim \pi,
  \end{align*}
  where \(\pi\) is a prior distribution over \(\phi\). Then, if the Bayesian model is regular in the sense of Definition~\ref{def:regularity}, the posterior distribution of \(\phi\) concentrates around \(\phi^{*}\):
  \begin{equation*}
    \pi \left( \phi \in U \mid D_{1:n} \right) \overset{P_{\phi^{*}}}{\to} 1,
  \end{equation*}
  for every neighborhood \(U\) of \(\phi^{*}\), where \(P_{\phi^{*}}\) is the probability measure induced by \(f_{\phi^{*}}\).
\end{theorem}

\section{Derivation of the Bayes Factor for the Validation Threshold}
\label{app:BF_derivation}

Under the model described in Section~\ref{subsec:alternative_threshold} of the main paper, the marginal likelihoods under the null and alternative hypotheses can be explicitly stated as follows:
\begin{align*}
    L\!\left(\widehat{V}_{S} \mid H_{0}\right) & = \binom{n}{n \, \widehat{V}_{S}} \, \left( \frac{1}{2} \right)^{n \, \widehat{V}_{S}} \, \left( 1 - \frac{1}{2} \right)^{n - n \, \widehat{V}_{S}} \\
    & = \binom{n}{n \, \widehat{V}_{S}} \, \frac{1}{2^{n}}, \\
    L\!\left(\widehat{V}_{S} \mid H_{1}\right) & = \int_{\frac{1}{2}}^{1} \binom{n}{n \, \widehat{V}_{S}} \, v_{S}^{n \, \widehat{V}_{S}} \, (1 - v_{S})^{n - n \, \widehat{V}_{S}} \frac{v_{S}^{a-1} \, (1 - v_{S})^{b-1}}{B(a,b) \, (1 - F_{\mathrm{Beta}(a,b)}\!\left(\tfrac{1}{2}\right))} \, \text{d}v_{S} \\
    & = \binom{n}{n \, \widehat{V}_{S}} \frac{B\!\left(a + n \, \widehat{V}_{S}, \, b + n - n \, \widehat{V}_{S}\right) \, \left(1 - F_{\mathrm{Beta}\left(a + n \, \widehat{V}_{S},\, b + n - n \, \widehat{V}_{S}\right)}\!\!\left(\tfrac{1}{2}\right)\right)}{B(a,b) \, \left(1 - F_{\mathrm{Beta}(a,b)}\!\left(\tfrac{1}{2}\right)\right)},
\end{align*}
where \( L\!\left(\widehat{V}_{S} \mid H_{0}\right) \) and \( L\!\left(\widehat{V}_{S} \mid H_{1}\right) \) are the marginal likelihoods under the null and alternative hypotheses, respectively, \(B(a, b)\) is the Beta function and \(F_{\mathrm{Beta}(a,b)}\) is the CDF of the \(\mathrm{Beta}(a,b)\) distribution. Then, the Bayes factor in favor of the alternative hypothesis is
\begin{align*}
    BF_{n} & = \frac{L\!\left(\widehat{V}_{S} \mid H_{1}\right)}{L\!\left(\widehat{V}_{S} \mid H_{0}\right)} \\
    & = \frac{B\!\left(a + n \, \widehat{V}_{S},\, b + n - n \, \widehat{V}_{S}\right) \, \left(1 - F_{\mathrm{Beta}\left(a + n \, \widehat{V}_{S},\, b + n - n \, \widehat{V}_{S}\right)}\!\!\left(\tfrac{1}{2}\right)\right)}{B(a,b) \, \left(1 - F_{\mathrm{Beta}(a,b)}\!\left(\tfrac{1}{2}\right)\right)} \, 2^{n}.
\end{align*}
Notice that the distribution of \(BF_{n}\) under \(H_{0}\) has closed form: since \(n \, \widehat{V}_{S} \in \{0, \ldots, n\}\), the Bayes factor takes at most \(n + 1\) distinct values \(x_{0}, \ldots, x_{n}\) with
\begin{equation*}
    \mathbb{P}(BF_{n} = x_{k} \mid H_0) = \binom{n}{k} \frac{1}{2^{n}}, \quad k = 0, \ldots, n,
\end{equation*}
where
\begin{equation*}
    x_{k} = \frac{B(a + k,\, b + n - k) \, \left(1 - F_{\mathrm{Beta}(a+k,\, b+n-k)}\!\left(\tfrac{1}{2}\right)\right)}{B(a,b) \, \left(1 - F_{\mathrm{Beta}(a,b)}\!\left(\tfrac{1}{2}\right)\right)} \, 2^{n}, \quad k = 0, \ldots, n.
\end{equation*}
Finally, to compute the probability on the left-hand side of Equation~\eqref{eq:V_S_power}, we observe that
\begin{equation*}
    \mathbb{P} \left( BF_{n} = x_{k} \mid V_{S} = v_{S} > \frac{1}{2} \right) = \binom{n}{k} v_{S}^{k}(1-v_{S})^{n-k}, \quad k = 0, \ldots, n.
\end{equation*}

\section{Simulation Study Details}

This appendix describes the DGPs and the Markov chain Monte Carlo implementation details for the simulation study in Section~\ref{sec:simulation_study} of the main paper.

\subsection{DGPs}
\label{app:data_generating_processes}

In all six settings, the treatment indicator \(Z_i \iid \text{Bernoulli}\left(\frac{1}{2}\right)\) is drawn independently of all potential outcomes. The DGPs for \(P_i\) in each setting are as follows.

\begin{itemize}[leftmargin=*]
    \item \textbf{Setting 1 (Useless surrogate):} The potential outcomes are generated as follows:
    \begin{equation*}
    \begin{array}{l}
        Y_{1i} \iid \mathcal{N}(5, 3), \quad S_{1i} \iid \mathcal{N}\!\left(\tfrac{2}{3}, 1\right), \\
        Y_{0i} \iid \mathcal{N}(3, 3), \quad S_{0i} \iid \mathcal{N}\!\left(\tfrac{2}{3}, 1\right),
    \end{array}
    \quad i = 1, \ldots, n.
    \end{equation*}
    Therefore, by the properties of Gaussian distributions, we have that
    \begin{equation*}
        P_{i} \iid \mathcal{N}_4\!\left(\mu^*, \Sigma^*\right), \quad i = 1, \ldots, n,
    \end{equation*}
    where
    \begin{align*}
        \mu^* &= \left(5,\, \tfrac{2}{3},\, 3,\, \tfrac{2}{3}\right)^\top, &
        \Sigma^* &= \mathrm{diag}\!\left(3,\, 1,\, 3,\, 1\right).
    \end{align*}

    \item \textbf{Setting 2 (Perfect surrogate):} The potential outcomes are generated as follows:
    \begin{equation*}
    \begin{array}{l}
        Y_{1i} \iid \mathcal{N}(6, 3), \quad S_{1i} \mid Y_{1i} \indepsim \mathcal{N}(Y_{1i}, 0.1), \\
        Y_{0i} \iid \mathcal{N}(2.5, 3), \quad S_{0i} \mid Y_{0i} \indepsim \mathcal{N}(Y_{0i}, 0.1),
    \end{array}
    \quad i = 1, \ldots, n.
    \end{equation*}
    Therefore, by the properties of Gaussian distributions, we have that
    \begin{equation*}
        P_{i} \iid \mathcal{N}_4\!\left(\mu^*, \Sigma^*\right), \quad i = 1, \ldots, n,
    \end{equation*}
    where
    \begin{align*}
        \mu^* &= (6,\, 6,\, 2.5,\, 2.5)^\top, &
        \Sigma^* &= I_2 \otimes \begin{bmatrix} 3 & 3 \\ 3 & 3.1 \end{bmatrix}.
    \end{align*}
    where \(\otimes\) denotes the Kronecker product.
    \item \textbf{Setting 3 (Imperfect surrogate):} The potential outcomes are generated as follows:
    \begin{equation*}
    \begin{array}{l}
        S_{1i} \iid \mathcal{N}(5, 3), \quad Y_{1i} \mid S_{1i} \indepsim \mathcal{N}(S_{1i} + 1.5,\, 0.6), \\
        S_{0i} \iid \mathcal{N}(3, 3), \quad Y_{0i} \mid S_{0i} \indepsim \mathcal{N}(S_{0i},\, 0.6),
    \end{array}
    \quad i = 1, \ldots, n.
    \end{equation*}
    Therefore, by the properties of Gaussian distributions, we have that
    \begin{equation*}
        P_{i} \iid \mathcal{N}_4\!\left(\mu^*, \Sigma^*\right), \quad i = 1, \ldots, n,
    \end{equation*}
    where
    \begin{align*}
        \mu^* &= (6.5,\, 5,\, 3,\, 3)^\top, &
        \Sigma^* &= I_2 \otimes \begin{bmatrix} 3.6 & 3 \\ 3 & 3 \end{bmatrix}.
    \end{align*}
    \item \textbf{Setting 4 (Misspecified model):} The potential outcomes are generated as follows:
    \begin{equation*}
    \begin{array}{rl}
        S_{1i} & \iid \text{LogNormal}(2.5,\, 1.5^2), \\
        Y_{1i} \mid S_{1i} & \indepsim 2.0 + 1.2\sqrt{S_{1i}} + 0.3\,e^{\frac{S_{1i}}{500}} + \text{LogNormal}(0,\, 0.3), \\
        S_{0i} & \iid \text{LogNormal}(0.5,\, 1.5^2), \\
        Y_{0i} \mid S_{0i} & \indepsim 0.8\sqrt{S_{0i}} + 0.2\,e^{\frac{S_{0i}}{50}} + \text{LogNormal}(0,\, 0.3),
    \end{array}
    \quad i = 1, \ldots, n.
    \end{equation*}
    \item \textbf{Setting 5 (Binary covariate):} The potential outcomes are generated as follows:
    \begin{equation*}
    \begin{array}{rl}
        P_{i} \mid X_{i} & \indepsim X_{i}\,\mathcal{N}_{4}\!\left(\mu^{1}, \Sigma^{1}\right) + (1 - X_{i})\,\mathcal{N}_{4}\!\left(\mu^{0}, \Sigma^{0}\right), \\
        X_i & \iid \text{Bernoulli} \left( \frac{1}{2} \right),
    \end{array}
    \quad i = 1, \ldots, n,
    \end{equation*}
    where
    \begin{align*}
        \mu^{0} &= (5, 5, 0, 0)^\top, &
        \mu^{1} &= (5, -5, 0, -10)^\top, \\
        \Sigma^{0} &= I_2 \otimes \begin{bmatrix} 1 & 1 \\ 1 & 2 \end{bmatrix}, &
        \Sigma^{1} &= I_2 \otimes \begin{bmatrix} 1 & 1 \\ 1 & 2 \end{bmatrix},
    \end{align*}
    \item \textbf{Setting 6 (Gaussian covariate):} The potential outcomes are generated as follows:
    \begin{equation*}
    \begin{array}{rl}
        P_{i} \mid X_{i} & \indepsim \mathcal{N}_{4}\!\left(X_{i}\,\beta,\, \Sigma\right), \\
        X_i & \iid \mathcal{N}(m,\, v^2),
    \end{array}
    \quad i = 1, \ldots, n,
    \end{equation*}
    where
    \begin{align*}
        \beta &= (1,\, 7,\, 0,\, 6)^\top, &
        \Sigma &= \tfrac{1}{2}\, I_{4}, \\
        m &= 3, &
        v^2 &= 1.
    \end{align*}
\end{itemize}
